FILE-ENTRIES=./dics/Diccionario.txt

# TO-LOWERCASE
# Se utilizan los textos en minuscula

# PreProcess
# Opcional
# Si no se comenta entonces se hará el preprocesamiento.
# -s => Search
# -r => Replace

PREPROCESS-RULE= -s "(\\D)([\\.,])(\\D)?" -r "$1 $2 $3"

# DICTIONARIES

# EXACT-DICTIONARY
# Opcional
# -c => CaseSensitive = true.
# -a => AllMatches = true

# RULE-BASED-DICTIONARY
# Opcional
# Si no se comenta entonces se incluirá este diccionario.
# -s => Search
# -m => Means

RULE-BASED-DICTIONARY-RULE= -s "[A-Za-z0-9](([_\.\-]?[a-zA-Z0-9]+)*)@([A-Za-z0-9]+)(([\.\-]?[a-zA-Z0-9]+)*)\.([A-Za-z]{2,})" -m "Mail"
RULE-BASED-DICTIONARY-RULE= -s "[0-9]+" -m "numero"
RULE-BASED-DICTIONARY-RULE= -s "\sy+" -m "y"
RULE-BASED-DICTIONARY-RULE= -s "\bal(?=\s)" -m "al"
RULE-BASED-DICTIONARY-RULE= -s "\bl(a)+(?=\s)" -m "la"
RULE-BASED-DICTIONARY-RULE= -s "esquina" -m "esquina"
RULE-BASED-DICTIONARY-RULE= -s "entre" -m "entre"
RULE-BASED-DICTIONARY-RULE= -s "casi" -m "casi"
RULE-BASED-DICTIONARY-RULE= -s "de" -m "de"
